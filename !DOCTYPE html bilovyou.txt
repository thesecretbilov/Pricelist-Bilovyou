<script>
  const cart = [];
  const cartItemsEl = document.getElementById('cart-items');
  const cartTotalEl = document.getElementById('cart-total');
  const orderBtn = document.getElementById('order-btn');
  const toggleCartBtn = document.getElementById('toggle-cart');
  const cartContent = document.getElementById('cart-content');

  // Fungsi format angka ke Rp
  function formatRp(num) {
    return 'Rp ' + num.toString().replace(/\B(?=(\d{3})+(?!\d))/g, '.');
  }

  // Update tampilan keranjang
  function updateCart() {
    cartItemsEl.innerHTML = '';
    let total = 0;
    cart.forEach((item, index) => {
      total += item.price;
      const li = document.createElement('li');
      li.innerHTML = `<span class="name">${item.name}</span> <span>${formatRp(item.price)}</span> <button class="remove-btn" data-index="${index}">&times;</button>`;
      cartItemsEl.appendChild(li);
    });
    cartTotalEl.textContent = 'Total: ' + formatRp(total);
  }

  // Tambah item ke keranjang
  function addToCart(name, price) {
    cart.push({name, price});
    updateCart();
  }

  // Event klik item harga untuk tambah ke keranjang
  document.querySelectorAll('.price-list li').forEach(li => {
    li.addEventListener('click', () => {
      const name = li.getAttribute('data-name');
      const price = parseInt(li.getAttribute('data-price'));
      addToCart(name, price);
      alert(`"${name}" berhasil ditambahkan ke keranjang!`);
    });
  });

  // Event klik tombol hapus item di keranjang
  cartItemsEl.addEventListener('click', e => {
    if(e.target.classList.contains('remove-btn')) {
      const index = e.target.getAttribute('data-index');
      cart.splice(index, 1);
      updateCart();
    }
  });

  // Event klik tombol order via WA
  orderBtn.addEventListener('click', () => {
    if(cart.length === 0) {
      alert('Keranjang kosong! Silakan pilih produk terlebih dahulu.');
      return;
    }
    let message = 'Halo, saya ingin memesan produk berikut:%0A';
    cart.forEach(item => {
      message += `- ${item.name} : Rp ${item.price.toLocaleString('id-ID')}%0A`;
    });
    message += `Total: Rp ${cart.reduce((a,b) => a + b.price, 0).toLocaleString('id-ID')}`;
    const waLink = `https://wa.me/6285252460755?text=${message}`;
    window.open(waLink, '_blank');
  });

  // Toggle keranjang (buka/tutup)
  let cartOpen = true;
  toggleCartBtn.addEventListener('click', () => {
    cartOpen = !cartOpen;
    if(cartOpen) {
      cartContent.style.display = 'block';
      toggleCartBtn.innerHTML = '&#9650;';
    } else {
      cartContent.style.display = 'none';
      toggleCartBtn.innerHTML = '&#9660;';
    }
  });
</script>
